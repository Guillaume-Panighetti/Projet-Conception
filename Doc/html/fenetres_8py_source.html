560 def plateau_cartes(self):
561 """
562 @brief Affiche le plateau de cartes.
563
564 @return list: Liste des cartes.
565 """
566 # Définir la taille et l'espacement des cartes
567 taille_carte = (self.ecran_l // 11, self.ecran_h // 5)
568 espacement = 20
569 marge = 20
570
571 # Calculer la position de départ du plateau de cartes
572 x_debut = self.ecran_l / 5
573 y_debut = self.ecran_h - (2 * taille_carte[1] + espacement) - marge
574
575 self.noms_cartes = ['0', '1', '2', '3', '5', '8', '13', '20', '40', '100', 'interro', 'cafe']
576
577 self.liste_cartes = []
578 for i, nom_carte in enumerate(self.noms_cartes):
579 x = x_debut + (i % 6) * (taille_carte[0] + espacement)
580 y = y_debut + (i // 6) * (taille_carte[1] + espacement)
581
582 carte = Cartes(nom_carte, (x, y), self.fenetrefenetrefenetrefenetrefenetrefenetrefenetre, taille_carte)
583 self.liste_cartes.append(carte)
584
585 for carte in self.liste_cartes:
586 carte.afficher()
